```sql

```

### 1c. Vérifier les FK après renommage

```sql
-- Vérifier que les FK pointent toujours correctement
SELECT
    tc.table_name,
    kcu.column_name,
    ccu.table_name AS foreign_table
FROM information_schema.table_constraints tc
JOIN information_schema.key_column_usage kcu ON tc.constraint_name = kcu.constraint_name
JOIN information_schema.constraint_column_usage ccu ON ccu.constraint_name = tc.constraint_name
WHERE tc.constraint_type = 'FOREIGN KEY'
AND tc.table_schema = 'public'
ORDER BY tc.table_name;
```

Toutes les FK doivent pointer vers les bonnes tables. Corriger si nécessaire.

### Node SESSION_LOAD
- Type : postgres v2.6

- Requête :
```sql
SELECT
    u.utilisateur_id,
    u.telegram_id,
    p.projet_id,
    p.hashtag AS projet_actif,
    p.nom AS projet_nom,
    p.brief_executif,
    p.statut AS projet_statut,
    s.session_id,
    s.pole_actif,
    s.etat AS session_etat,
    s.resume_session,

FROM utilisateurs u

    AND s.etat = 'ouvert'
    AND s.last_activity > NOW() - INTERVAL '24 hours'
LEFT JOIN projets p ON p.projet_id = s.projet_id

    SELECT jsonb_agg(
        jsonb_build_object('role', m.role, 'contenu', m.contenu, 'pole', m.pole_emetteur, 'ts', m.created_at)

    FROM (
        SELECT role, contenu, pole_emetteur, created_at
        FROM messages
        WHERE session_id = s.session_id
```

### Node CERVEAU_XO (Code JS — routing déterministe)

```javascript
const msg = $input.first().json;
const text = (msg.message?.text || '').trim();
const sessionData = $('SESSION_LOAD').first().json;

// Extraire hashtags
const hashtags = (text.match(/#(\w+)/g) || []).map(h => h.slice(1).toLowerCase());
const poleHashtags = ['finance', 'marketing', 'strategie', 'produit', 'risk'];
const polesActives = hashtags.filter(h => poleHashtags.includes(h));
const projetHashtag = hashtags.find(h => !poleHashtags.includes(h));

// Détecter le mode
let route = 'dialogue';
let mode = 'dialogue';

if (text.startsWith('/menu')) { route = 'menu'; }
else if (text.startsWith('/reset') || text === 'reset') { route = 'reset'; }
else if (!sessionData.session_id && !text.startsWith('/')) { route = 'nouvelle_idee'; mode = 'qualification'; }
else if (text.toLowerCase().match(/j.ai une id[eé]e|et si on|je veux cr[eé]er|je veux d[eé]velopper/)) {
    route = 'nouvelle_idee'; mode = 'qualification';
}
else { route = 'dialogue'; mode = 'dialogue'; }

return [{
    json: {
        ...msg,
        routing: {
            route,
            mode,
            poles_actives: polesActives.length > 0 ? polesActives : null,
            projet_hashtag: projetHashtag || sessionData.projet_actif || null,
            session_context: sessionData,
            text_clean: text.replace(/#\w+/g, '').trim()
        }
    }
}];
```

### Node DB_WRITE
- Type : postgres v2.6
- Deux requêtes séquentielles :

**Requête 1 — Créer session si inexistante :**
```sql
INSERT INTO sessions (projet_id, utilisateur_id, pole_actif, etat)
SELECT
    p.projet_id,
    u.utilisateur_id,
    'XO',
    'ouvert'
FROM utilisateurs u
LEFT JOIN projets p ON p.hashtag = '{{ $('CERVEAU_XO').item.json.routing.projet_hashtag }}'
    AND p.utilisateur_id = u.utilisateur_id
WHERE u.telegram_id = {{ $('TELEGRAM_INPUT').item.json.message.chat.id }}
AND NOT EXISTS (
    SELECT 1 FROM sessions s2
    WHERE s2.utilisateur_id = u.utilisateur_id
    AND s2.etat = 'ouvert'
    AND s2.last_activity > NOW() - INTERVAL '24 hours'
)
RETURNING session_id;
```

**Requête 2 — Insérer les deux messages (user + assistant) :**
```sql
INSERT INTO messages (session_id, projet_id, role, pole_emetteur, contenu, modele_utilise)
SELECT s.session_id, s.projet_id, role, pole, contenu, modele
FROM sessions s
JOIN utilisateurs u ON u.utilisateur_id = s.utilisateur_id
CROSS JOIN (VALUES
    ('user', NULL, '{{ $('CERVEAU_XO').item.json.routing.text_clean }}', NULL),
    ('assistant', 'XO', '{{ $('XO_AGENT').item.json.output }}', 'claude-sonnet-4-20250514')
) AS msgs(role, pole, contenu, modele)
WHERE u.telegram_id = {{ $('TELEGRAM_INPUT').item.json.message.chat.id }}
AND s.etat = 'ouvert'
AND s.last_activity > NOW() - INTERVAL '24 hours'
ORDER BY s.last_activity DESC
LIMIT 1;

-- Update compteur et timestamp
UPDATE sessions SET
    nb_messages = nb_messages + 2,
    last_activity = NOW()
WHERE utilisateur_id = (
    SELECT utilisateur_id FROM utilisateurs
    WHERE telegram_id = {{ $('TELEGRAM_INPUT').item.json.message.chat.id }}
)
AND etat = 'ouvert'
AND last_activity > NOW() - INTERVAL '24 hours';
```
